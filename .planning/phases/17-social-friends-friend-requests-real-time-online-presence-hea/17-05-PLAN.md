### Task 2: Friend search

Search input with debounced fetch to `GET /api/friends/search?q=`:

```jsx
function handleSearch(q) {
  setSearchQuery(q);
  if (q.length < 2) { setSearchResults([]); return; }
  clearTimeout(searchTimer.current);
  searchTimer.current = setTimeout(() => {
    fetch("/api/friends/search?q=" + encodeURIComponent(q))
      .then(r => r.json())
      .then(setSearchResults);
  }, 300);
}
```

Each result shows: name + win rate + "➕ Add Friend" button.
On add: POST `/api/friends/request` → show "Sent ✓" toast.

### Task 4: Challenge receive popup (modal)

Global component `ChallengePopup` rendered in App when a challenge is received:

```jsx
function ChallengePopup({ challenge, onAccept, onDecline }) {
  const [timeLeft, setTimeLeft] = useState(60);
  
  useEffect(() => {
    const iv = setInterval(() => {
      const rem = Math.max(0, Math.ceil((challenge.expiresAt - Date.now()) / 1000));
      setTimeLeft(rem);
      if (rem <= 0) { onDecline(); clearInterval(iv); }
    }, 1000);
    return () => clearInterval(iv);
  }, [challenge]);
  
  return (
    <div className="challenge-overlay">
      <div className="challenge-popup">
        <div className="challenge-icon">⚔️</div>
        <div className="challenge-text">
          {t("challenge.received", { name: challenge.from.displayName })}
        </div>
        {challenge.stake > 0 && (
          <div className="challenge-stake">💰 {challenge.stake} coin</div>
        )}
        <div className="challenge-timer">⏱️ {timeLeft}s</div>
        <div className="challenge-actions">
          <button className="btn primary" onClick={onAccept}>
            ✓ {t("challenge.accept")}
          </button>
          <button className="btn ghost" onClick={onDecline}>
            ✗ {t("challenge.decline")}
          </button>
        </div>
      </div>
    </div>
  );
}
```

- Rendered as full-screen overlay (like existing leave-confirm modal)
- Cannot be dismissed by clicking outside — must choose accept/decline
- Auto-decline when timer hits 0

```javascript
  // Sender: challenge was accepted, enter room (already seated by server)

  // Show "Challenge declined" notice

```

Accept handler:
```javascript
function handleAcceptChallenge() {
  socket.emit("friend:challenge-accept", { challengeId: incomingChallenge.challengeId }, (res) => {
```
