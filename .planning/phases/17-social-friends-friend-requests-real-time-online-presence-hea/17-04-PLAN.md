# Plan 17-04: Direct Challenge Flow

## Objective

Implement the server-side challenge system: sender picks a friend + stake → server creates a room → emits challenge to recipient → recipient accepts/declines → game starts or room is destroyed.

### Task 2: Socket event — friend:challenge (sender)

```javascript
socket.on("friend:challenge", async ({ friendId, stake }, cb) => {
  const userId = socket.data.userId;
  if (!userId) return cb && cb({ ok: false, code: "AUTH_REQUIRED" });
  
  // Validate: target is an accepted friend
  const friendship = await getFriendshipStatus(userId, friendId);
  if (friendship !== 'accepted') return cb && cb({ ok: false, code: "NOT_FRIENDS" });
  
  // Validate: target is online (not in-game, not offline)
  const targetPresence = userPresence.get(friendId);
  if (targetPresence !== 'online') return cb && cb({ ok: false, code: "FRIEND_NOT_AVAILABLE" });
  
  // Validate: sender not already in a game/room
  if (socket.data.code) return cb && cb({ ok: false, code: "ALREADY_IN_ROOM" });
  
  // Validate stake
  const validStake = VALID_STAKES.includes(stake) ? stake : 0;
  
  // If wagered: validate sender balance
  if (validStake > 0) {
    const debitRef = "challenge_host_" + Date.now() + "_" + socket.id;
    const debitResult = await debitWallet(userId, validStake, "wager_debit", debitRef);
    if (!debitResult.ok) return cb && cb({ ok: false, code: "INSUFFICIENT_BALANCE" });
    socket.emit("balanceUpdate", { balance: debitResult.balance });
    var hostWagerId = debitRef;
  }
  
  // Create room (reuse existing room shape)
  const code = newCode();
  rooms[code] = {
    code, players: {}, order: [], started: false, turn: null, scores: {},
    lastStarter: null, mode: "classic", turnTimer: null, turnDeadline: null,
    resolving: false, lastActivityAt: Date.now(),
    stake: validStake, hostWagerId: hostWagerId || null,
    challengeRoom: true, // flag to identify challenge rooms
    purchases: {},
  };
  
  // Seat the sender
  const clientId = socket.data.clientId || socket.id;
  rooms[code].players[clientId] = {
    sid: socket.id, ready: false, occ: null, hits: new Set(), online: true,
    timer: null, bonus: 0, profile: sanitizeProfile(/* sender profile */),
    userId, wagerId: hostWagerId || null,
    inv: newInv(), decoyCell: null,
  };
  rooms[code].order.push(clientId);
  socket.join(code);
  socket.data.code = code;
  socket.data.clientId = clientId;
  
  // Create pending challenge
  const challengeId = "ch_" + Date.now() + "_" + userId;
  const expiresAt = Date.now() + 60000;
  const timer = setTimeout(() => {
    // Expire: destroy room, refund sender, notify both
    expireChallenge(challengeId);
  }, 60000);
  
  pendingChallenges.set(challengeId, {
    senderId: userId, receiverId: friendId,
    roomCode: code, stake: validStake,
    expiresAt, timer, hostWagerId: hostWagerId || null,
    senderSocketId: socket.id,
  });
  
  // Emit challenge to recipient's sockets
  const recipientSockets = userSockets.get(friendId);
  if (recipientSockets) {
    for (const sid of recipientSockets) {
      io.to(sid).emit("friend:challenge-received", {
        challengeId, 
        from: { id: userId, displayName: /* lookup */ },
        stake: validStake,
        expiresAt,
      });
    }
  }
  
  cb && cb({ ok: true, challengeId, code });
});
```

### Task 3: Socket event — friend:challenge-accept (recipient)

```javascript
socket.on("friend:challenge-accept", async ({ challengeId }, cb) => {
  const userId = socket.data.userId;
  const challenge = pendingChallenges.get(challengeId);
  if (!challenge) return cb && cb({ ok: false, code: "CHALLENGE_NOT_FOUND" });
  if (challenge.receiverId !== userId) return cb && cb({ ok: false, code: "NOT_YOUR_CHALLENGE" });
  if (Date.now() > challenge.expiresAt) return cb && cb({ ok: false, code: "CHALLENGE_EXPIRED" });
  
  // If wagered: debit recipient
  if (challenge.stake > 0) {
    const debitRef = "challenge_join_" + Date.now() + "_" + socket.id;
    const debitResult = await debitWallet(userId, challenge.stake, "wager_debit", debitRef);
    if (!debitResult.ok) return cb && cb({ ok: false, code: "INSUFFICIENT_BALANCE" });
    socket.emit("balanceUpdate", { balance: debitResult.balance });
    var joinerWagerId = debitRef;
  }
  
  // Clear the challenge timer
  clearTimeout(challenge.timer);
  pendingChallenges.delete(challengeId);
  
  // Join recipient to the room (same as joinRoom logic)
  const room = rooms[challenge.roomCode];
  if (!room) return cb && cb({ ok: false, code: "ROOM_NOT_FOUND" });
  
  const clientId = socket.data.clientId || socket.id;
  room.players[clientId] = {
    sid: socket.id, ready: false, occ: null, hits: new Set(), online: true,
    timer: null, bonus: 0, profile: sanitizeProfile(/* recipient profile */),
    userId, wagerId: joinerWagerId || null,
    inv: newInv(), decoyCell: null,
  };
  room.order.push(clientId);
  socket.join(challenge.roomCode);
  socket.data.code = challenge.roomCode;
  socket.data.clientId = clientId;
  
  // Emit standard room events (same as joinRoom)
  io.to(challenge.roomCode).emit("roomUpdate", roomPublic(room));
  io.to(challenge.roomCode).emit("opponentJoined");
  // Exchange profiles
  const oppId = opponentOf(room, clientId);
  if (oppId) {
    emitToClient(room, oppId, "oppProfile", room.players[clientId].profile || null);
    emitToClient(room, clientId, "oppProfile", room.players[oppId].profile || null);
  }
  
  cb && cb({ ok: true, code: challenge.roomCode, stake: challenge.stake });
  
  // Notify sender that challenge was accepted
  io.to(challenge.senderSocketId).emit("friend:challenge-accepted", { challengeId });
});
```

### Task 4: Socket event — friend:challenge-decline (recipient)

```javascript
socket.on("friend:challenge-decline", async ({ challengeId }, cb) => {
  const userId = socket.data.userId;
  const challenge = pendingChallenges.get(challengeId);
  if (!challenge || challenge.receiverId !== userId) return cb && cb({ ok: false });
  
  // Clear timer, destroy room, refund sender
  clearTimeout(challenge.timer);
  pendingChallenges.delete(challengeId);
  
  // Refund sender wager if any
  if (challenge.stake > 0 && challenge.hostWagerId) {
    creditWallet(challenge.senderId, challenge.stake, "wager_refund", challenge.hostWagerId + "_declined").catch(() => {});
    // Push updated balance to sender
    const senderSock = io.of("/").sockets.get(challenge.senderSocketId);
    if (senderSock) {
      getWalletBalance(challenge.senderId).then(b => senderSock.emit("balanceUpdate", { balance: b })).catch(() => {});
    }
  }
  
  // Destroy the challenge room
  const room = rooms[challenge.roomCode];
  if (room) {
    // Remove sender from room
    const senderSock = io.of("/").sockets.get(challenge.senderSocketId);
    if (senderSock) {
      senderSock.leave(challenge.roomCode);
      senderSock.data.code = null;
    }
    delete rooms[challenge.roomCode];
  }
  
  // Notify sender
  io.to(challenge.senderSocketId).emit("friend:challenge-declined", { challengeId });
  
  cb && cb({ ok: true });
});
```

### Task 5: expireChallenge helper

```javascript
function expireChallenge(challengeId) {
  const challenge = pendingChallenges.get(challengeId);
  if (!challenge) return;
  pendingChallenges.delete(challengeId);
  
  // Refund sender wager
  if (challenge.stake > 0 && challenge.hostWagerId) {
    creditWallet(challenge.senderId, challenge.stake, "wager_refund", challenge.hostWagerId + "_expired").catch(() => {});
    const senderSock = io.of("/").sockets.get(challenge.senderSocketId);
    if (senderSock) {
      getWalletBalance(challenge.senderId).then(b => senderSock.emit("balanceUpdate", { balance: b })).catch(() => {});
    }
  }
  
  // Destroy room
  const room = rooms[challenge.roomCode];
  if (room) {
    const senderSock = io.of("/").sockets.get(challenge.senderSocketId);
    if (senderSock) {
      senderSock.leave(challenge.roomCode);
      senderSock.data.code = null;
    }
    delete rooms[challenge.roomCode];
  }
  
  // Notify both
  io.to(challenge.senderSocketId).emit("friend:challenge-expired", { challengeId });
  const recipientSockets = userSockets.get(challenge.receiverId);
  if (recipientSockets) {
    for (const sid of recipientSockets) {
      io.to(sid).emit("friend:challenge-expired", { challengeId });
    }
  }
}
```

### Task 6: Cleanup on sender disconnect

If the sender disconnects while a challenge is pending → auto-expire the challenge (refund, destroy room, notify recipient):

```javascript
// In disconnect handler, after presence cleanup:
for (const [challengeId, ch] of pendingChallenges) {
  if (ch.senderSocketId === socket.id) {
    expireChallenge(challengeId);
  }
}
```

## Success Criteria

- [ ] Sender can challenge an online friend with stake selection
- [ ] Recipient receives challenge popup via Socket.IO
- [ ] Accept → both in room → placement phase starts (same as joinRoom)
- [ ] Decline → room destroyed, sender refunded, both notified
- [ ] Timeout (60s) → same as decline
- [ ] Cannot challenge in-game or offline friends (server rejects)
- [ ] Wagered challenges: debit/refund flow works correctly
- [ ] Sender disconnect → auto-expire
