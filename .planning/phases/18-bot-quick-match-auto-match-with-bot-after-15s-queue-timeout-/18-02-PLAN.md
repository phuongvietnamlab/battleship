```javascript
// bot-engine.js — Server-side bot game logic for Phase 18 (Bot Quick Match)
// Pure functions for ship placement and targeting. No React, no DOM.

const BOARD = 11;
const FLEET_SIZES = [5, 4, 3, 3, 2];

function key(r, c) { return r + ',' + c; }

function cellsFor(r, c, size, dir) {
  const cells = [];
  for (let i = 0; i < size; i++) {
    cells.push(dir === 'h' ? { r, c: c + i } : { r: r + i, c });
  }
  return cells;
}

function inBounds(cells) {
  return cells.every(x => x.r >= 0 && x.r < BOARD && x.c >= 0 && x.c < BOARD);
}

/**
 * Generate a valid random fleet placement.
 * Returns { occ: Set<key>, ships: Array<Set<key>> }
 */
function generateBotFleet() {
  const occ = new Set();
  const ships = [];
  for (const size of FLEET_SIZES) {
    let ok = false, t = 0;
    while (!ok && t++ < 800) {
      const d = Math.random() < 0.5 ? 'h' : 'v';
      const r = Math.floor(Math.random() * BOARD);
      const c = Math.floor(Math.random() * BOARD);
      const cells = cellsFor(r, c, size, d);
      if (inBounds(cells) && cells.every(x => !occ.has(key(x.r, x.c)))) {
        const set = new Set();
        cells.forEach(x => { const k = key(x.r, x.c); occ.add(k); set.add(k); });
        ships.push(set);
        ok = true;
      }
    }
  }
  return { occ, ships };
}

/**
 * Pick a random unshot cell (pure random targeting).
 * Returns cell key string "r,c" or null.
 */
function pickRandomTarget(shots) {
  const pool = [];
  for (let r = 0; r < BOARD; r++) {
    for (let c = 0; c < BOARD; c++) {
      const k = key(r, c);
      if (!shots.has(k)) pool.push(k);
    }
  }
  return pool.length ? pool[Math.floor(Math.random() * pool.length)] : null;
}

/**
 * Get a random delay for bot firing (2000-5000ms).
 */
function getBotFireDelay() {
  return 2000 + Math.floor(Math.random() * 3000);
}

module.exports = { generateBotFleet, pickRandomTarget, getBotFireDelay, BOARD, FLEET_SIZES, key };
```

// ─── Phase 18: Bot Quick Match ───────────────────────────────────────────────

const { generateBotFleet, pickRandomTarget, getBotFireDelay } = require('./bot-engine');

// Track which bots are currently in active matches
const activeBotMatches = new Map(); // botUserId → roomCode

async function createBotMatchRoom(humanEntry, botUser, type) {
  const code = newCode();
  const botFleet = generateBotFleet();
  const stake = type === 'wagered' ? (humanEntry.stake || 0) : 0;

  // Debit bot wallet for the stake (if wagered)
  let botWagerId = null;
  if (stake > 0) {
    const refId = 'bot_wager_' + Date.now() + '_' + botUser.id;
    const result = await debitWallet(botUser.id, stake, 'wager_debit', refId);
    if (!result.ok) {
      // Bot can't afford → free match
      // Refund human's stake since we're downgrading to free
      if (humanEntry.wagerId) {
        await creditWallet(humanEntry.userId, humanEntry.stake, 'wager_refund', humanEntry.wagerId + '_botfail');
        const sock = io.of('/').sockets.get(humanEntry.socketId);
        if (sock) {
          const bal = await getWalletBalance(humanEntry.userId);
          sock.emit('balanceUpdate', { balance: bal });
        }
      }
      // Retry as free match with stake=0
      return createBotMatchRoom(humanEntry, botUser, 'free');
    }
    botWagerId = refId;
  }

  // Create room
  const botClientId = 'bot_' + botUser.id + '_' + Date.now();
  rooms[code] = {
    code, players: {}, order: [], started: false, turn: null, scores: {},
    lastStarter: null, mode: 'classic',
    turnTimer: null, turnDeadline: null,
    resolving: false, lastActivityAt: Date.now(),
    matchQueueType: type,
    stake,
    purchases: {},
    isBot: true, // Phase 18: flag for bot match
    botUserId: botUser.id,
    botClientId,
    botFleet, // { occ, ships }
    botShots: new Set(),
    matchStartedAt: new Date(),
  };

  // Seat human player
  rooms[code].players[humanEntry.clientId] = {
    sid: humanEntry.socketId, ready: false, occ: null, hits: new Set(), online: true,
    timer: null, bonus: 0,
    profile: humanEntry.profile,
    userId: humanEntry.userId ?? null,
    matchQueueType: type,
    wagerId: humanEntry.wagerId ?? null,
    inv: newInv(), decoyCell: null,
  };
  rooms[code].order.push(humanEntry.clientId);

  // Seat bot player (virtual — no real socket)
  rooms[code].players[botClientId] = {
    sid: null, ready: true, occ: botFleet.occ, hits: new Set(), online: true,
    timer: null, bonus: 0,
    profile: { name: botUser.display_name, avatar: botUser.avatar_url },
    userId: botUser.id,
    matchQueueType: type,
    wagerId: botWagerId,
    inv: newInv(), decoyCell: null,
  };
  rooms[code].order.push(botClientId);

  // Join human socket to room
  const humanSocket = io.of('/').sockets.get(humanEntry.socketId);
  if (humanSocket) {
    humanSocket.join(code);
    humanSocket.data.code = code;
    humanSocket.data.clientId = humanEntry.clientId;
    humanSocket.data.queueType = null;
    humanSocket.data.queueClientId = null;
  }

  // Track active bot
  activeBotMatches.set(botUser.id, code);

  // Emit matchFound to human
  if (humanSocket) {
    humanSocket.emit('matchFound', { code, stake });
    humanSocket.emit('oppProfile', {
      name: botUser.display_name,
      avatar: botUser.avatar_url,
    });
    humanSocket.emit('roomUpdate', roomPublic(rooms[code]));
    humanSocket.emit('opponentJoined');
  }

  // Bot is already "ready" (ships placed) — emit oppReady when human places
  upsertGuestCredential(humanEntry.clientId); // fire-and-forget

  console.log(`[bot-match] created room ${code}: ${humanEntry.clientId} vs bot ${botUser.display_name} (stake=${stake})`);
}

### Task 3: Bot turn logic — firing and turn handling

/**
 * Execute bot's turn in a bot match room.
 * Called when it's the bot's turn (after human misses, or at game start if bot goes first).
 */
function scheduleBotTurn(code) {
  const room = rooms[code];
  if (!room || !room.isBot || room.resolving) return;

  const delay = getBotFireDelay();
  setTimeout(() => {
    if (!rooms[code] || rooms[code].turn !== room.botClientId) return;
    executeBotShot(code);
  }, delay);
}

function executeBotShot(code) {
  const room = rooms[code];
  if (!room || !room.isBot) return;

  const target = pickRandomTarget(room.botShots);
  if (!target) return; // no cells left (shouldn't happen)

  room.botShots.add(target);
  const [r, c] = target.split(',').map(Number);

  // Find human player
  const humanClientId = room.order.find(id => id !== room.botClientId);
  const humanPlayer = room.players[humanClientId];
  if (!humanPlayer || !humanPlayer.occ) return;

  const hit = humanPlayer.occ.has(target);

  // Apply hit tracking
  if (hit) {
    room.players[room.botClientId].hits.add(target);
  }

  // Check if ship sunk
  let sunkShip = null;
  // (sunk detection uses same logic as doShot — check all ship cells hit)

  // Emit shot result to human
  const humanSocket = io.of('/').sockets.get(humanPlayer.sid);
  if (humanSocket) {
    humanSocket.emit('shotResult', { // or whatever event the client expects for incoming fire
      r, c, hit,
      shooter: room.botClientId,
    });
  }

  // Check win condition
  // If all human ships sunk → bot wins
  const allSunk = checkAllShipsSunk(humanPlayer, room.players[room.botClientId]);

    endBotMatch(code, room.botUserId, humanPlayer.userId, 'normal');

  if (hit) {
    scheduleBotTurn(code);
  } else {
    // Miss → human's turn
    room.turn = humanClientId;

    if (humanSocket) {
      humanSocket.emit('turnUpdate', { turn: humanClientId });
    }
  }
}

### Task 4: Bot match end handling

```javascript
async function endBotMatch(code, winnerId, loserId, reason) {
  const room = rooms[code];
  if (!room) return;

  // Record match (same as PvP)
  await recordMatch(winnerId, loserId, reason, room.mode || 'classic', room.matchStartedAt, room.stake);

  // Clean up bot tracking
  activeBotMatches.delete(room.botUserId);

  // Emit game over to human
  const humanClientId = room.order.find(id => id !== room.botClientId);
  const humanPlayer = room.players[humanClientId];
  const humanSocket = humanPlayer ? io.of('/').sockets.get(humanPlayer.sid) : null;

  const humanWon = winnerId === humanPlayer?.userId;
  if (humanSocket) {
    humanSocket.emit('gameOver', { win: humanWon, reason });
    // Send updated balance
    if (humanPlayer.userId) {
      const balance = await getWalletBalance(humanPlayer.userId);
      humanSocket.emit('balanceUpdate', { balance });
    }
  }

  // Destroy room
  delete rooms[code];
}
```
